74   /**
75    * Method to flatten customer's offer with weekly informations.
76    * Added new columns begin_week_number.begin_date,end_week_number,end_date
77    * @param offerFlat
78    * @param numberTab
79    * @return customerOfferCall
80    */
81   def customerOfferCall(offerFlat:DataFrame,numberTab:DataFrame) : DataFrame = {
82     val offer = offerFlat.withColumn("begin_week_number", date_format(col("begin_date"), "w").cast(IntegerType))
83       .withColumn("end_week_number", date_format(col("end_date"), "w").cast(IntegerType))
84 
85     val customerOfferCall = offer.join(numberTab, numberTab("week_number")
86       .between(offer("begin_week_number"), offer("end_week_number")))
87       .select("week_number", "offer_id", "offer_name", "cust_id", "begin_date", "end_date", "begin_week_number", "end_week_number")
88     customerOfferCall
89   }
